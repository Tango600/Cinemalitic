Declare=TP_FILM_ID;
SetValue=TP_FILM_ID=:FILM_ID;
SetValue=IS_CLOSED_PERIOD;SQL=select count(1) from CLOSED_PERIODS P where P.CLOSED = false and (select C.AGREEMENT_DATE from CONTRACTS C where C.CONTRACT_ID = :CONTRACT_ID) between P.PERIOD_BEGIN and P.PERIOD_END;
if &IS_CLOSED_PERIOD=0 then
ExecQuery=update FILMS_TO_CONTRACTS set EXCLUDE_FROM_DELIVERIES=1 where CONTRACT_ID = &CONTRACT_ID and FILM_ID = &TP_FILM_ID;
Else
Message=Период закрыт!;
EndIf
